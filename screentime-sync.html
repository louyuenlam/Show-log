// ═══════════════════════════════════════════════════════════════════════════════ // Screentime Sync — Google Apps Script // Full CRUD for: Entries, Ongoing, Wishlist, Watching (cache) // v2: Strict field whitelisting + type validation to prevent scrambled writes // ═══════════════════════════════════════════════════════════════════════════════ const SHEETS = { ENTRIES: { name: 'Entries', headers: ['id','title','type','platform','season','episode','episodeTo','totalEp','date','notes','rating','updatedAt'] }, ONGOING: { name: 'Ongoing', headers: ['id','title','completedType','platform','season','currentEp','totalEp','airingStatus','completed','completedDate','givenUp','givenUpDate','givenUpReason','score','review','notes','posterUrl','updatedAt'] }, WISHLIST: { name: 'Wishlist', headers: ['id','title','type','platform','notes','added','updatedAt'] }, UPCOMING: { name: 'Upcoming', headers: ['id','title','type','platform','season','releaseDate','releaseYear','airDay','firstAirDate','totalEp','notes','updatedAt'] }, WATCHING: { name: 'Watching', headers: ['id','title','season','totalEp','platform','currentEp','completedType'] }, META: { name: 'Meta', headers: ['key','value'] }, }; // ── Field type validators for ENTRIES ───────────────────────────────────────── // If a value lands in the wrong column (e.g. "Gimy" in totalEp, "3" in platform), // validation rejects the write rather than corrupting the sheet. const ENTRIES_FIELD_TYPES = { id: v => v === '' || v === null || v === undefined || (typeof v === 'number') || /^\d+$/.test(String(v).trim()), title: v => typeof v === 'string' && v.length > 0 && v.length < 500, type: v => v === 'TV Show' || v === 'Movie' || v === '', platform: v => v === '' || v === null || v === undefined || (typeof v === 'string' && !/^\d+$/.test(v.trim())), season: v => v === '' || v === null || v === undefined || /^\d+$/.test(String(v).trim()), episode: v => v === '' || v === null || v === undefined || /^\d+$/.test(String(v).trim()), episodeTo: v => v === '' || v === null || v === undefined || /^\d+$/.test(String(v).trim()), totalEp: v => v === '' || v === null || v === undefined || /^\d+$/.test(String(v).trim()), date: v => v === '' || v === null || v === undefined || (typeof v === 'string' && /^\d{4}-\d{2}-\d{2}/.test(v)), notes: v => v === '' || v === null || v === undefined || typeof v === 'string', rating: v => v === '' || v === null || v === undefined || (typeof v === 'number' && v >= 0 && v <= 5) || /^[0-5]?$/.test(String(v).trim()), }; function validateEntry(entry) { const issues = []; for (const field in ENTRIES_FIELD_TYPES) { const v = entry[field]; if (!ENTRIES_FIELD_TYPES[field](v)) { issues.push(field + '=' + JSON.stringify(v)); } } return issues; } // ── Sheet helpers ───────────────────────────────────────────────────────────── function getSheet(key) { const cfg = SHEETS[key]; const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(cfg.name); if (!sheet) { sheet = ss.insertSheet(cfg.name); sheet.appendRow(cfg.headers); sheet.setFrozenRows(1); sheet.getRange(1,1,1,cfg.headers.length).setFontWeight('bold').setBackground('#f0e8dc'); } return sheet; } function formatDate(v) { if(!v) return ''; if(v instanceof Date) { return v.getFullYear() + '-' + String(v.getMonth()+1).padStart(2,'0') + '-' + String(v.getDate()).padStart(2,'0'); } const s = String(v); // Already YYYY-MM-DD if(/^\d{4}-\d{2}-\d{2}$/.test(s)) return s; // ISO with time — strip time part if(/^\d{4}-\d{2}-\d{2}T/.test(s)) return s.slice(0,10); return s; } function sheetToObjects(sheet) { const data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; const headers = data[0]; const DATE_FIELDS = new Set(['date','completedDate','givenUpDate','added','updatedAt']); return data.slice(1).map(row => { const obj = {}; headers.forEach((h,i) => { const v = row[i]; if(v === '' || v === null || v === undefined) { obj[h] = ''; return; } // Format date fields as YYYY-MM-DD to avoid timezone issues in browsers if(DATE_FIELDS.has(h) && v instanceof Date) { obj[h] = formatDate(v); return; } obj[h] = v; }); return obj; }); } function findRowById(sheet, id) { const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][0]) === String(id)) return i + 1; // 1-indexed } return -1; } function objectToRow(obj, headers) { return headers.map(h => obj[h] !== undefined ? obj[h] : ''); } function now() { return new Date().toISOString(); } // ── Meta (last modified timestamp) ─────────────────────────────────────────── function getLastModified() { const sheet = getSheet('META'); const rows = sheetToObjects(sheet); const row = rows.find(r => r.key === 'lastModified'); return row ? row.value : ''; } function setLastModified() { const sheet = getSheet('META'); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === 'lastModified') { sheet.getRange(i+1, 2).setValue(now()); return; } } sheet.appendRow(['lastModified', now()]); } // ── doGet — read all data or check lastModified ─────────────────────────────── function doGet(e) { try { const type = (e.parameter && e.parameter.type) || 'all'; if (type === 'lastModified') { return jsonResponse({ lastModified: getLastModified() }); } if (type === 'watching') { return jsonResponse({ watching: sheetToObjects(getSheet('WATCHING')) }); } if (type === 'entries') { return jsonResponse({ entries: sheetToObjects(getSheet('ENTRIES')) }); } if (type === 'ongoing') { return jsonResponse({ ongoing: sheetToObjects(getSheet('ONGOING')) }); } if (type === 'wishlist') { return jsonResponse({ wishlist: sheetToObjects(getSheet('WISHLIST')) }); } return jsonResponse({ entries: sheetToObjects(getSheet('ENTRIES')), ongoing: sheetToObjects(getSheet('ONGOING')), wishlist: sheetToObjects(getSheet('WISHLIST')), upcoming: sheetToObjects(getSheet('UPCOMING')), lastModified: getLastModified() }); } catch(err) { return jsonResponse({ error: err.message }); } } // ── doPost — write operations ───────────────────────────────────────────────── function doPost(e) { try { const body = JSON.parse(e.postData.contents); const op = body.op; if (body.type === 'watching' || body.type === 'removeWatching') { return handleLegacyWatching(body); } if (Array.isArray(body)) { return handleLegacyEntries(body); } const sheet = getSheet(body.sheet); const headers = SHEETS[body.sheet].headers; if (op === 'upsert') { // Validate ENTRIES writes — reject if any field is in the wrong shape if (body.sheet === 'ENTRIES') { const issues = validateEntry(body.data); if (issues.length > 0) { return jsonResponse({ error: 'Validation failed: ' + issues.join('; '), received: body.data }); } } const item = { ...body.data, updatedAt: now() }; const rowNum = findRowById(sheet, item.id); if (rowNum > 0) { sheet.getRange(rowNum, 1, 1, headers.length).setValues([objectToRow(item, headers)]); } else { sheet.appendRow(objectToRow(item, headers)); } if (body.sheet !== 'WATCHING') { rebuildWatching(); setLastModified(); } return jsonResponse({ success: true, op: 'upsert', id: item.id }); } if (op === 'batchUpsert') { const items = body.data.map(d => ({ ...d, updatedAt: now() })); const existingData = sheet.getDataRange().getValues(); const existingIds = new Set(existingData.slice(1).map(r => String(r[0]))); const newRows = items.filter(item => !existingIds.has(String(item.id))); newRows.forEach(item => sheet.appendRow(objectToRow(item, headers))); if (body.sheet === 'ONGOING') { rebuildWatching(); } if (body.sheet !== 'WATCHING') { setLastModified(); } return jsonResponse({ success: true, op: 'batchUpsert', count: newRows.length }); } if (op === 'delete') { const rowNum = findRowById(sheet, body.id); if (rowNum > 0) sheet.deleteRow(rowNum); if (body.sheet !== 'WATCHING') { rebuildWatching(); setLastModified(); } return jsonResponse({ success: true, op: 'delete', id: body.id }); } if (op === 'replaceAll') { const lastRow = sheet.getLastRow(); if (lastRow > 1) { for (let r = lastRow; r >= 2; r -= 100) { const s = Math.max(2, r - 99); sheet.deleteRows(s, r - s + 1); } } const items = body.data.map(d => ({ ...d, updatedAt: now() })); items.forEach(item => sheet.appendRow(objectToRow(item, headers))); if (body.sheet !== 'WATCHING') { rebuildWatching(); setLastModified(); } return jsonResponse({ success: true, op: 'replaceAll', count: items.length }); } return jsonResponse({ error: 'Unknown op: ' + op }); } catch(err) { return jsonResponse({ error: err.message }); } } // ── Legacy handlers (phone log compatibility) ───────────────────────────────── function handleLegacyEntries(items) { const sheet = getSheet('ENTRIES'); const headers = SHEETS.ENTRIES.headers; const existingData = sheet.getDataRange().getValues(); const existingIds = new Set(existingData.slice(1).map(r => String(r[0]))); let saved = 0; const rejected = []; items.forEach(entry => { if (existingIds.has(String(entry.id))) return; // Validate before writing — reject scrambled entries const issues = validateEntry(entry); if (issues.length > 0) { rejected.push({ id: entry.id, title: entry.title, issues }); return; } sheet.appendRow(objectToRow({ ...entry, updatedAt: now() }, headers)); saved++; }); if (saved > 0) { rebuildWatching(); setLastModified(); } return jsonResponse({ success: true, saved, rejected }); } function handleLegacyWatching(body) { if (body.type === 'removeWatching') { const sheet = getSheet('WATCHING'); const rows = sheet.getDataRange().getValues(); const titleLc = String(body.title||'').trim().toLowerCase(); const season = String(body.season||''); for (let i = rows.length - 1; i >= 1; i--) { if (String(rows[i][1]||'').trim().toLowerCase() === titleLc && String(rows[i][2]||'') === season) { sheet.deleteRow(i + 1); } } return jsonResponse({ success: true }); } if (body.type === 'watching') { rebuildWatching(); return jsonResponse({ success: true }); } } // ── Rebuild Watching cache from Ongoing ─────────────────────────────────────── function rebuildWatching() { const ongoing = sheetToObjects(getSheet('ONGOING')); const watchSheet = getSheet('WATCHING'); const headers = SHEETS.WATCHING.headers; const isTrue = (v) => { if (v === true) return true; const s = String(v).toLowerCase(); return s === 'true' || s === 'yes'; }; const active = ongoing.filter(o => !isTrue(o.completed) && !isTrue(o.givenUp) && o.completedType !== 'Movie' ); const lastRow = watchSheet.getLastRow(); if (lastRow > 1) { watchSheet.getRange(2, 1, lastRow - 1, headers.length).clearContent(); } active.forEach((o, i) => { watchSheet.getRange(i + 2, 1, 1, 7).setValues([[ o.id, o.title, o.season, o.totalEp, o.platform, o.currentEp, o.completedType || 'TV Show' ]]); }); } function jsonResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON);
